def _current_week_monday() -> str:
    now = datetime.now(timezone.utc)
    monday = now - timedelta(days=now.weekday())
    return monday.strftime("%Y-%m-%d")


def record_signal(signal: str, score: float) -> None:
    week = _current_week_monday()
    with stats_lock:
        if weekly_stats["week_start"] != week:
            weekly_stats["signals"]    = []
            weekly_stats["week_start"] = week
        weekly_stats["signals"].append({
            "signal": signal,
            "score":  round(score, 3),
            "ts":     datetime.now(timezone.utc).strftime("%a %H:%M"),
        })

def calc_ema(series: pd.Series, span: int) -> pd.Series:
    return series.ewm(span=span, adjust=False).mean()


def calc_rsi(series: pd.Series, period: int = 14) -> pd.Series:
    delta = series.diff()
    gain  = delta.clip(lower=0).rolling(period).mean()
    loss  = (-delta.clip(upper=0)).rolling(period).mean()
    rs    = gain / loss.replace(0, np.nan)
    return 100 - (100 / (1 + rs))


def calc_macd(series: pd.Series, fast: int = 12, slow: int = 26, signal: int = 9):
    ema_fast    = calc_ema(series, fast)
    ema_slow    = calc_ema(series, slow)
    macd_line   = ema_fast - ema_slow
    signal_line = calc_ema(macd_line, signal)
    histogram   = macd_line - signal_line
    return macd_line, signal_line, histogram


def calc_atr(df: pd.DataFrame, period: int = 14) -> pd.Series:
    hl  = df["High"] - df["Low"]
    hpc = (df["High"] - df["Close"].shift()).abs()
    lpc = (df["Low"]  - df["Close"].shift()).abs()
    tr  = pd.concat([hl, hpc, lpc], axis=1).max(axis=1)
    return tr.rolling(period).mean()


def add_indicators(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    df["ema20"]  = calc_ema(df["Close"], 20)
    df["ema50"]  = calc_ema(df["Close"], 50)
    df["ema200"] = calc_ema(df["Close"], 200)
    df["rsi"]    = calc_rsi(df["Close"])
    df["macd"], df["macd_signal"], df["macd_hist"] = calc_macd(df["Close"])
    df["atr"]    = calc_atr(df)
    return df

def _is_high_impact_gold_event(title: str, country: str, impact: str) -> bool:
    if country.upper() not in ("USD", "US"):
        return False
    if impact.lower() != "high":
        return False
    return any(kw in title.lower() for kw in _CALENDAR_KEYWORDS)


def fetch_economic_calendar() -> list[dict]:
    events = []
    headers = {
        "User-Agent": "Mozilla/5.0 (compatible; GoldBot/1.0)",
        "Accept":     "application/json",
    }
    for url in _CALENDAR_URLS:
        try:
            r = requests.get(url, headers=headers, timeout=15)
            if not r.ok:
                continue
            for ev in r.json():
                title   = ev.get("title", "")
                country = ev.get("country", "")
                impact  = ev.get("impact", "")
                if _is_high_impact_gold_event(title, country, impact):
                    events.append({
                        "title":    title,
                        "date":     ev.get("date", ""),
                        "country":  country,
                        "impact":   impact,
                        "forecast": ev.get("forecast") or "—",
                        "previous": ev.get("previous") or "—",
                        "actual":   ev.get("actual") or None,
                    })
        except Exception as e:
            log.warning(f"[calendar] {url}: {e}")
    seen, unique = set(), []
    for ev in events:
        key = (ev["title"], ev["date"])
        if key not in seen:
            seen.add(key)
            unique.append(ev)
    unique.sort(key=lambda x: x["date"])
    return unique[:10]


def format_calendar_for_prompt(events: list[dict]) -> str:
    if not events:
        return "لا توجد أحداث اقتصادية كبرى مجدولة هذا الأسبوع."
    lines = []
    for ev in events:
        try:
            dt  = datetime.fromisoformat(ev["date"].replace("Z", "+00:00"))
            day = dt.strftime("%a %d %b %H:%M UTC")
        except Exception:
            day = ev["date"][:16]
        actual_str = f" | Actual: {ev['actual']}" if ev["actual"] else ""
        lines.append(
            f"- {day} | {ev['title']} "
            f"(Forecast: {ev['forecast']} | Prev: {ev['previous']}{actual_str})"
        )
    return "\n".join(lines)


def format_calendar_for_telegram(events: list[dict]) -> str:
    if not events:

            f"    🗓 *{day}* — {ev['title']}"

        )
    return "\n".join(lines) + "\n"

def score_timeframe(df: pd.DataFrame) -> dict:
    row  = df.iloc[-1]
    prev = df.iloc[-2]
    scores = {}

    rsi = row["rsi"]
    if rsi >= 70:        scores["rsi"] = -0.8
    elif rsi <= 30:      scores["rsi"] = +0.8
    elif rsi > 55:       scores["rsi"] = +0.4
    elif rsi < 45:       scores["rsi"] = -0.4
    else:                scores["rsi"] = 0.0

    hist_now  = row["macd_hist"]
    hist_prev = prev["macd_hist"]
    macd_cross_bull = (prev["macd"] < prev["macd_signal"] and
                       row["macd"]  > row["macd_signal"])
    macd_cross_bear = (prev["macd"] > prev["macd_signal"] and
                       row["macd"]  < row["macd_signal"])
    if macd_cross_bull:                          scores["macd"] = +1.0
    elif macd_cross_bear:                        scores["macd"] = -1.0
    elif hist_now > 0 and hist_now > hist_prev:  scores["macd"] = +0.5
    elif hist_now < 0 and hist_now < hist_prev:  scores["macd"] = -0.5
    elif hist_now > 0:                           scores["macd"] = +0.2
    elif hist_now < 0:                           scores["macd"] = -0.2
    else:                                        scores["macd"] = 0.0

    price  = row["Close"]
    ema20  = row["ema20"]
    ema50  = row["ema50"]
    ema200 = row["ema200"]
    if   price > ema20 > ema50 > ema200: scores["ema"] = +1.0
    elif price > ema20 > ema50:          scores["ema"] = +0.6
    elif price > ema20:                  scores["ema"] = +0.3
    elif price < ema20 < ema50 < ema200: scores["ema"] = -1.0
    elif price < ema20 < ema50:          scores["ema"] = -0.6
    elif price < ema20:                  scores["ema"] = -0.3
    else:                                scores["ema"] = 0.0

    composite = float(np.mean(list(scores.values())))
    return {
        "score":           round(composite, 3),
        "rsi":             round(float(rsi), 1),
        "rsi_score":       scores["rsi"],
        "macd_score":      scores["macd"],
        "ema_score":       scores["ema"],
        "macd_cross_bull": macd_cross_bull,
        "macd_cross_bear": macd_cross_bear,
        "price":           round(float(row["Close"]), 2),
        "ema20":           round(float(ema20), 2),
        "ema50":           round(float(ema50), 2),
        "ema200":          round(float(ema200), 2),
        "atr":             round(float(row["atr"]), 2),
        "high":            round(float(row["High"]), 2),
        "low":             round(float(row["Low"]), 2),
        "volume":          int(row["Volume"]),
    }


def get_confluence(tf_scores: dict) -> float:
    weighted_sum = sum(
        data["score"] * TIMEFRAMES[tf]["weight"]
        for tf, data in tf_scores.items()
    )
    total_weight = sum(TIMEFRAMES[tf]["weight"] for tf in tf_scores)
    return round(weighted_sum / total_weight, 3) if total_weight else 0.0


def classify_score(score: float) -> tuple[str, str]:
    if score >= CONFLUENCE_THRESHOLD:
        return "BUY", "🟢"
    elif score <= -CONFLUENCE_THRESHOLD:
        return "SELL", "🔴"
    return "HOLD", "🟡"


# ═══════════════════════════════════════════════════════════════════════════════
# RISK MANAGEMENT ENGINE
# ═══════════════════════════════════════════════════════════════════════════════

def calc_risk_params(entry: float, atr: float, signal: str) -> dict:
    sl_distance  = round(atr * ATR_SL_MULT, 2)
    tp2_distance = round(sl_distance * MIN_RR, 2)

    if signal == "BUY":
        sl  = round(entry - sl_distance, 2)
        tp1 = round(entry + sl_distance, 2)
        tp2 = round(entry + tp2_distance, 2)
    elif signal == "SELL":
        sl  = round(entry + sl_distance, 2)
        tp1 = round(entry - sl_distance, 2)
        tp2 = round(entry - tp2_distance, 2)
    else:
        sl = tp1 = tp2 = entry

    risk_usd      = ACCOUNT_SIZE * RISK_PCT
    position_size = round(risk_usd / sl_distance, 4) if sl_distance > 0 else 0.0

    return {
        "entry":         entry,
        "sl":            sl,
        "tp1":           tp1,
        "tp2":           tp2,
        "sl_distance":   sl_distance,
        "tp2_distance":  tp2_distance,
        "rr_ratio":      round(tp2_distance / sl_distance, 1) if sl_distance else 0,
        "risk_usd":      round(risk_usd, 2),
        "position_size": position_size,
    }

# AI INTERFACE  (OpenRouter — free-tier fallback chain)
# ═══════════════════════════════════════════════════════════════════════════════

def call_openrouter(prompt: str) -> str:
    url     = "https://openrouter.ai/api/v1/chat/completions"
    headers = {
        "Authorization": f"Bearer {OPENROUTER_API_KEY}",
        "Content-Type":  "application/json",
    }
    for model in OPENROUTER_MODELS:
        payload = {"model": model,
                   "messages": [{"role": "user", "content": prompt}]}
        for attempt in range(1, 3):
            try:
                r      = requests.post(url, json=payload, headers=headers, timeout=60)
                result = r.json()
                if "choices" in result:
                    log.info(f"AI answered via: {model}")
                    return result["choices"][0]["message"]["content"]
                error = result.get("error", {})
                if error.get("code") == 429:
                    wait = error.get("metadata", {}).get("retry_after_seconds", 20)
                    log.warning(f"Rate-limited ({model}), retry in {wait}s [{attempt}/2]")
                    time.sleep(wait + 2)
                else:
                    log.warning(f"Error ({model}): {error.get('message')} → next model")
                    break
            except Exception as e:
                log.warning(f"Connection error ({model}): {e}")
                time.sleep(10)
    return "سمح لي، وقع خطأ ف الاتصال بالـ AI."


def score_news_sentiment(news: str) -> float:
    if not news:
        return 0.0
    prompt = (
        "You are a quantitative analyst specializing in Gold (XAUUSD). "
        "Analyze the following news headlines and return ONLY a single decimal number "
        "between -1.0 (strongly bearish for gold) and +1.0 (strongly bullish for gold). "
        f"No explanation, just the number.\n\nHeadlines:\n{news}"
    )
    try:
        raw = call_openrouter(prompt).strip()
        val = float("".join(c for c in raw if c in "0123456789.-")[:6])
        return max(-1.0, min(1.0, val))
    except Exception:
        return 0.0

def weekend_report() -> str:
    now            = datetime.now(timezone.utc)
    days_to_monday = (7 - now.weekday()) % 7 or 7
    monday_open    = now.replace(hour=0, minute=5, second=0, microsecond=0) + timedelta(days=days_to_monday)
    hours_left     = round((monday_open - now).total_seconds() / 3600, 1)
    day_name       = "السبت" if now.weekday() == 5 else "الأحد"
    return (
        f"📅 *تقرير نهاية الأسبوع — الذهب (XAUUSD)*\n"
        f"_{now.strftime('%Y-%m-%d %H:%M')} UTC_\n\n"
        f"🔴 *السوق مغلق اليوم ({day_name})*\n\n"
        "⏳ سيعاود السوق فتح أبوابه الاثنين الصباح.\n"
        f"🕐 الوقت المتبقي حتى الافتتاح: `~{hours_left}` ساعة\n\n"
        "💡 *استغل عطلة الأسبوع لـ:*\n"

            direction_label = "صاعد 📈"

    log_lines = ""
    for s in signals[-10:]:
        emoji = "🟢" if s["signal"] == "BUY" else "🔴" if s["signal"] == "SELL" else "🟡"
        log_lines += f"    {emoji} {s['ts']} — {s['signal']} (`{s['score']:+.3f}`)\n"
    if not log_lines:
        log_lines = "    _لا توجد إشارات مسجلة_\n"
